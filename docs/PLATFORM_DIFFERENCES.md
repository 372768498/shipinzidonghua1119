```javascript
function calculateViralScore(video, account) {
  const baseThreshold = {
    tiktok: 1000000,
    youtube: 500000
  }
  
  // 根据账号粉丝数调整阈值
  const followerMultiplier = {
    "0-1000": 0.01,      // 新号：1万播放可能就是爆款
    "1000-10000": 0.05,   // 小号：5万播放
    "10000-100000": 0.1,  // 中号：10万播放
    "100000-1000000": 0.5, // 大号：50万播放
    "1000000+": 1.0       // 超大号：100万播放
  }
  
  const adjustedThreshold = baseThreshold[video.platform] 
    * getMultiplier(account.followers)
  
  return video.views >= adjustedThreshold
}
```
